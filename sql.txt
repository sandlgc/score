CREATE TABLE student_answer_records (

    record_id INT AUTO_INCREMENT PRIMARY KEY,
    -- 学生 ID，用于关联学生信息
    student_id INT NOT NULL,
    -- 答题章节
    chapter VARCHAR(255) NOT NULL,
    -- 回答题目数
    answered_questions INT NOT NULL,
    -- 答对题目数
    correct_questions INT NOT NULL,
    -- 连续答对最高记录
    consecutive_correct_max INT NOT NULL,
    -- 正确率，以百分比形式存储
    accuracy DECIMAL(5, 2) NOT NULL,
);

CREATE TABLE student_answer_records (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    chapter VARCHAR(255) NOT NULL,
    -- 回答题目数
    answered_questions INT NOT NULL,
    -- 答对题目数
    correct_questions INT NOT NULL,
    accuracy DECIMAL(5, 2) NOT NULL,
    consecutive_correct_max INT NOT NULL,
    wrong_question_ids TEXT,
    attempt_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    answer_time INT NOT NULL DEFAULT 0 COMMENT '答题总时间(秒)'
);
这个表包含以下字段：
id：记录的唯一标识符，自增。
student_id：学生的 ID。
chapter：答题的章节。
answered_questions 回答题目数
correct_questions 答对题目数
accuracy：答题的正确率，使用 DECIMAL(5, 2) 类型可以存储最多 5 位数字，其中 2 位是小数。
consecutive_correct_max：连续答对的最高记录。
wrong_question_ids：错题的 ID，使用 TEXT 类型可以存储多个 ID。

编写统计错题的 SQL
-- 统计每个学生的错题（曾经答错但后续未答对的题目）
INSERT INTO student_wrong_questions (student_id, chapter, wrong_question_ids)
SELECT 
    sar.student_id,
    sar.chapter,
    GROUP_CONCAT(DISTINCT wq.question_id) AS wrong_question_ids
FROM 
    student_answer_records sar
JOIN students s ON sar.student_id = s.student_id  -- 关联到students.student_id（字符串类型）
JOIN (
    -- 拆分wrong_question_ids中的题目ID（处理CSV格式）
    SELECT 
        student_id,
        chapter,
        SUBSTRING_INDEX(SUBSTRING_INDEX(sar.wrong_question_ids, ',', n.n), ',', -1) AS question_id
    FROM 
        student_answer_records sar
    JOIN (
        -- 生成足够的数字序列（根据实际错题数量调整）
        SELECT 1 AS n UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5
        UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL SELECT 9 UNION ALL SELECT 10
    ) n 
    ON CHAR_LENGTH(sar.wrong_question_ids) - CHAR_LENGTH(REPLACE(sar.wrong_question_ids, ',', '')) >= n.n - 1
    WHERE 
        sar.wrong_question_ids IS NOT NULL AND sar.wrong_question_ids != ''
) wq 
ON sar.student_id = wq.student_id AND sar.chapter = wq.chapter
WHERE 
    -- 排除后续答对的题目：检查是否存在后续记录中该题已不在错题列表
    NOT EXISTS (
        SELECT 1 
        FROM student_answer_records sar_later
        WHERE 
            sar_later.student_id = sar.student_id
            AND sar_later.chapter = sar.chapter
            AND sar_later.record_id > sar.record_id
            AND (sar_later.wrong_question_ids IS NULL OR 
                 FIND_IN_SET(wq.question_id, sar_later.wrong_question_ids) = 0)
    )
GROUP BY 
    sar.student_id, sar.chapter
ON DUPLICATE KEY UPDATE 
    wrong_question_ids = VALUES(wrong_question_ids),
    last_updated = NOW();

CREATE TABLE IF NOT EXISTS student_cards (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL COMMENT '学生学号',
    card_id INT NOT NULL COMMENT '对应core_values_cards.id',
    obtained_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '获得时间',
    UNIQUE KEY uk_student_card (student_id, card_id), -- 防止重复获得同一张
    FOREIGN KEY (student_id) REFERENCES students(student_id),
    FOREIGN KEY (card_id) REFERENCES core_values_cards(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT '学生核心价值观卡片收集表';
